String padding(字符串填充)

String.prototype.padStart 在开始部位填充，返回一个给出长度的字符串，填充物给定字符串，把字符串填充到期望的长度。

String.prototype.padEnd 从字符串的尾端右边开始填充，填充物给定字符串，把字符串填充到期望的长度。

console.log('hello'.padStart(20))
console.log('world'.padStart(20))
console.log('0.00'.padStart(20))
console.log('10,000.00'.padStart(20))   
console.log('250,000.00'.padStart(20))
console.log('ECMAScript'.padStart(20,'#'))
console.log('ECMAScript'.padStart(20,'#12345'))
console.log('hello'.padEnd(20))
console.log('world'.padEnd(20))
console.log('0.00'.padEnd(20))
console.log('10,000.00'.padEnd(20))   
console.log('250,000.00'.padEnd(20))
console.log('ECMAScript'.padEnd(20,'#'))
console.log('ECMAScript'.padEnd(20,'#12345'))
